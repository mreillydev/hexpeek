50// Distance to the next bound, exclusive of o
51#define distbound(o, b) ( ((o) % (b)) ? ((b) - (o) % (b)) : (b) )
52
53// Distance to the next bound, inclusive of o
54#define distbound_incl(o, b) ( ((o) % (b)) ? ((b) - (o) % (b)) : 0 )
55
56// Round o up to the next multiple of b (unless it already is a multiple of b)
57#define ceilbound(o, b) ( (o) + distbound_incl(o, b) )
